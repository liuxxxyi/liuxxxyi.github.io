严蔚敏LinkStack

#include <stdio.h>
#include <stdlib.h>
#define ERROR 0
#define OK 1
typedef int ElemType,Status;
typedef struct Node
{
ElemType data;
struct Node *next;
}Node;

typedef struct LinkStack
{
Node *top;
int size;
}LinkStack,*LinkStackP;

Status InitStack(LinkStackP *S) //给main中的指针s 分配一个LinkStack节点，即为LinkStack的结构；
{
(*S) = malloc(sizeof(LinkStack));
(*S)->top = NULL;
(*S)->size = 0;
return OK;
}

Status DesLinkStack(LinkStackP *S)
{
if(!S) return ERROR;
free(*S);
(*S) = NULL;
return OK;
}

Status ClearLinkStack(LinkStackP S)
{
if(!S) return ERROR;
S->top = NULL;
S->size = 0;
return OK;
}
Status Push(LinkStackP S,ElemType data)
{
if(!S) return ERROR;
Node *node = malloc(sizeof(Node));
node->data = data;
node->next = S->top;
S->top = node;
S->size++;
return OK;
}

Status Pop(LinkStackP S,ElemType *e)
{
if(!S || !S->top) return ERROR;
Node *node = S->top;
*e = node->data;
S->top = node->next;
S->size–;
free(node);
node = NULL;
printf(“pop is %d\n”,*e);
return OK;
}

int StackLen(LinkStackP S)
{
if(!S) return ERROR;
return S->size;
}

int GetElem(LinkStackP S)
{
if(S)
return S->size;
else
return ERROR;
}

int isEmpty(LinkStackP S)
{
if(!S) return ERROR;
else
{
if(S->size == 0)
return 1;
else
return 0;
}
}
